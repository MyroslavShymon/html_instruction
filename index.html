// Для браузера IE
if (document.all) {
x = event.x + document.body.scrollLeft;
y = event.y + document.body.scrollTop;
// Для остальных браузеров
} else {
var x = e.pageX/5; // Координата X курсора
var y = e.pageY/5; // Координата Y курсора
}
var element = document.getElementById("coords");
var clas_h = x+"px",clas_w = y+"px";
element.style.cssText="height:"+clas_h+"; width:"+clas_w+";";
}
